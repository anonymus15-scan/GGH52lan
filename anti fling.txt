	TextLabel_2.Text = "Detects players, Disables Collisions if suspected of fling, Please Give Some Time To Load. Rejoin to disable"

	TextLabel_2.TextSize = 14
	TextLabel_2.TextWrapped = true

	wait(7)

	ScreenGui:Destroy()
	
	local rotationThreshold = 100 -- Adjust this value based on your needs
	local transparencyValue = 0.5 -- Adjust this value to control the level of transparency
	local outlineColor = BrickColor.new("Bright red") -- You can change the color as needed

	local selectionBoxes = {} -- Dictionary to store SelectionBox objects for each target

	local function isPartRotatingTooFast(part)
		if part and part:IsA("BasePart") then
			local lastRotation = part.Rotation
			wait(0.1) -- Wait a short time to get a second rotation value for comparison
			local currentRotation = part.Rotation
			local rotationChange = (currentRotation - lastRotation).Magnitude
			return rotationChange > rotationThreshold
		end
		return false
	end

	local function isPartInLocalPlayerCharacter(part)
		local localPlayer = game.Players.LocalPlayer
		if localPlayer and localPlayer.Character then
			return part:IsDescendantOf(localPlayer.Character)
		end
		return false
	end

	local function createSelectionBox(target)
		local outline = Instance.new("SelectionBox")
		outline.Name = "RedOutline"
		outline.Color3 = outlineColor.Color
		outline.Adornee = target
		outline.Parent = target
		return outline
	end

	local function modifyRapidlyRotatingObjects()
		local objects = game.Workspace:GetDescendants()

		-- Update existing SelectionBoxes and remove non-existent targets
		for target, selectionBox in pairs(selectionBoxes) do
			if target:IsDescendantOf(game.Workspace) and not isPartInLocalPlayerCharacter(target) then
				if not isPartRotatingTooFast(target) then
					-- If the target is not rotating rapidly, remove the SelectionBox
					selectionBox:Destroy()
					selectionBoxes[target] = nil
				end
			else
				-- If the target is no longer in the workspace or part of the local player's character, remove the SelectionBox
				selectionBox:Destroy()
				selectionBoxes[target] = nil
			end
		end

		-- Add SelectionBoxes for new rapidly rotating objects
		for _, obj in pairs(objects) do
			if not isPartInLocalPlayerCharacter(obj) and isPartRotatingTooFast(obj) and not selectionBoxes[obj] then
				obj.Transparency = transparencyValue
				obj.CanCollide = false
				selectionBoxes[obj] = createSelectionBox(obj)
			end
		end
	end

	while true do
		modifyRapidlyRotatingObjects()
		wait(1) -- Adjust the time interval between checks (1 second in this example)
	end

	
end

local function onAvoidMethodClick()
	local teleportDistance = 25 -- Adjust this value to set the teleport distance in studs
	local transparencyThreshold = 50 -- Adjust this value to set the distance where the mesh becomes transparent
	local rotationThreshold = 100 -- Adjust this value based on your needs

	local function isObjectInCharacter(obj)
		local localPlayer = game.Players.LocalPlayer
		if localPlayer and localPlayer.Character then
			return obj:IsDescendantOf(localPlayer.Character)
		end
		return false
	end

	local function isPartRotatingTooFast(part)
		if part and part:IsA("BasePart") then
			local lastRotation = part.Rotation
			wait(0.1) -- Wait a short time to get a second rotation value for comparison
			local currentRotation = part.Rotation
			local rotationChange = (currentRotation - lastRotation).Magnitude
			return rotationChange > rotationThreshold
		end
		return false
	end

	local function teleportAwayFromPart(part)
		local localPlayer = game.Players.LocalPlayer
		if localPlayer and localPlayer.Character and part and part:IsA("BasePart") then
			if isPartRotatingTooFast(part) then
				local character = localPlayer.Character
				local characterRoot = character:FindFirstChild("HumanoidRootPart")
				if characterRoot then
					local distance = (characterRoot.Position - part.Position).Magnitude
					if distance <= teleportDistance then
						local direction = (characterRoot.Position - part.Position).Unit
						local teleportPosition = part.Position + direction * Vector3.new(teleportDistance, 0, teleportDistance)
						characterRoot.CFrame = CFrame.new(teleportPosition)
					end
				end
			end
		end
	end

	local function createTeleportRadius(part)
		local localPlayer = game.Players.LocalPlayer
		if localPlayer and localPlayer.Character and part and part:IsA("BasePart") then
			local radius = part:FindFirstChild("TeleportRadius")
			if isPartRotatingTooFast(part) then
				if not radius then
					radius = Instance.new("Part")
					radius.Name = "TeleportRadius"
					radius.Anchored = true
					radius.CanCollide = false
					radius.Transparency = 1
					radius.Size = Vector3.new(2 * teleportDistance, 0.2, 2 * teleportDistance) -- Twice the teleportDistance to cover each side
					radius.CFrame = CFrame.new(part.Position)
					local mesh = Instance.new("CylinderMesh")
					mesh.Scale = Vector3.new(1, 0.2, 1)
					mesh.Parent = radius
					radius.Parent = part
				end
				local localPlayer = game.Players.LocalPlayer
				local character = localPlayer.Character
				local characterRoot = character:FindFirstChild("HumanoidRootPart")
				if characterRoot then
					local direction = (characterRoot.Position - part.Position).Unit
					local lookVector = Vector3.new(direction.X, 0, direction.Z)
					radius.CFrame = CFrame.new(part.Position) * CFrame.Angles(0, math.atan2(lookVector.X, lookVector.Z), 0)
					local distance = (characterRoot.Position - part.Position).Magnitude
					if distance <= transparencyThreshold then
						radius.Transparency = 0.5
					else
						radius.Transparency = 1
					end
					local hue = ((distance - transparencyThreshold) / (teleportDistance - transparencyThreshold)) * 240
					local rainbowColor = Color3.fromHSV(hue / 360, 1, 1)
					radius.BrickColor = BrickColor.new(Color3.new(rainbowColor.R, rainbowColor.G, rainbowColor.B)) -- Convert the rainbow color to a BrickColor
				end
			else
				if radius then
					radius:Destroy()
				end
			end
		end
	end

	local function handleParts()
		local objects = game.Workspace:GetDescendants()

		for _, obj in pairs(objects) do
			if not isObjectInCharacter(obj) then
				teleportAwayFromPart(obj)
				createTeleportRadius(obj)
			end
		end
	end

	game:GetService("RunService").RenderStepped:Connect(function()
		handleParts()
	end)

end
